async function estimateBilling(){
  const payload = {
    mode: document.getElementById('mode').value,
    trade_type: document.getElementById('trade_type').value,
    trades_per_day: parseInt(document.getElementById('trades_per_day').value),
    duration: document.getElementById('duration').value,
    referral_count: 0
  };
  const r = await fetch('/api/billing/estimate', {method:'POST',headers:{'content-type':'application/json'}, body: JSON.stringify(payload)});
  const j = await r.json();
  document.getElementById('billingResult').innerText = "Estimated final bill: ₹" + j.final_bill + "  (details: base ₹"+j.base_price+", duration discount "+j.duration_discount_applied+"%, auto surcharge ₹"+j.auto_mode_surcharge+")";
}

async function makePrediction(){
  const stock = document.getElementById('stock').value;
  const when = document.getElementById('when').value || 'now';
  const res = await fetch('/api/predict', {method:'POST', headers:{'content-type':'application/json'}, body: JSON.stringify({stock, when})});
  const j = await res.json();
  const out = `
    <div><strong>Prediction for ${j.stock} at ${j.when}</strong></div>
    <div>Current (sim): ₹${j.current_price}</div>
    <div>Predicted range: ₹${j.pred_low} – ₹${j.pred_high}</div>
    <div>Confidence: ${j.confidence}%</div>
    <div class="small muted">(Astro bias applied hidden: ${j.astro_bias_applied})</div>
    <div style="margin-top:6px">
      <button onclick="savePlan('${j.stock}','${j.when}')">Save Trade Plan</button>
    </div>
  `;
  document.getElementById('predictionResult').innerHTML = out;
}

async function savePlan(stock, when){
  const payload = {
    user:'demo_user',
    mode: document.getElementById('mode').value,
    trade_type: document.getElementById('trade_type').value,
    trades_per_day: parseInt(document.getElementById('trades_per_day').value),
    duration: document.getElementById('duration').value,
    referral_count: 0
  };
  const r = await fetch('/api/subscribe', {method:'POST', headers:{'content-type':'application/json'}, body: JSON.stringify(payload)});
  const j = await r.json();
  alert('Subscribed & invoice generated: ₹' + j.invoice.final_bill);
}

async function startTraining(){
  await fetch('/api/training/start', {method:'POST'});
  document.getElementById('trainingStatus').innerText = 'Training started... check status.';
}

async function statusTraining(){
  const r = await fetch('/api/training/status');
  const j = await r.json();
  document.getElementById('trainingStatus').innerText = 'Progress: ' + j.progress + '% — ' + j.message;
}

async function submitFeedback(){
  const useful = document.getElementById('useful').value;
  const actual = document.getElementById('actual_price').value || null;
  const when = document.getElementById('when').value || 'now';
  const stock = document.getElementById('stock').value;
  const r = await fetch('/api/feedback', {method:'POST', headers:{'content-type':'application/json'}, body: JSON.stringify({stock, when, actual_price: actual, useful})});
  const j = await r.json();
  document.getElementById('fbmsg').innerText = 'Thanks — feedback saved.';
}
